### Step 4: Create DashboardPage (`frontend/src/pages/app/DashboardPage.tsx`)
**Time:** 60 minutes | **Test:** Dashboard functional with new design

```tsx
import { useState, useEffect } from 'react'
import { useNavigate } from 'react-router-dom'
import { supabase } from '../../lib/supabase'
import { FileText, Upload } from 'lucide-react'
import ResumeUpload from '../../components/ResumeUpload'

export default function DashboardPage() {
  const [user, setUser] = useState<any>(null)
  const [currentResume, setCurrentResume] = useState<any>(null)
  const [showUploadModal, setShowUploadModal] = useState(false)
  const navigate = useNavigate()

  useEffect(() => {
    loadUser()
  }, [])

  const loadUser = async () => {
    const { data: { user } } = await supabase.auth.getUser()
    if (user) {
      setUser(user)
      loadResume(user.id)
    }
  }

  const loadResume = async (userId: string) => {
    const { data } = await supabase
      .from('resumes')
      .select('*')
      .eq('user_id', userId)
      .order('created_at', { ascending: false })
      .limit(1)
      .single()

    if (data) setCurrentResume(data)
  }

  return (
    <div className="max-w-5xl">
      {/* Welcome Header */}
      <div className="mb-8">
        <h1 className="text-3xl font-bold text-[#333333] mb-2">
          Welcome back, {user?.email?.split('@')[0] || 'there'}! üëã
        </h1>
        <p className="text-[#666666]">
          Let's tailor your resume for your next opportunity.
        </p>
      </div>

      {/* Stats Cards Row */}
      <div className="grid grid-cols-2 gap-4 mb-8">
        {/* Tailorings Card */}
        <div className="card">
          <div className="text-sm text-[#666666] mb-1">Tailorings Used</div>
          <div className="text-3xl font-bold text-[#4CAF50]">1/5</div>
          <div className="text-xs text-[#666666] mt-2">4 remaining this month</div>
        </div>

        {/* Resumes Card */}
        <div className="card">
          <div className="text-sm text-[#666666] mb-1">Resume Slots</div>
          <div className="text-3xl font-bold text-[#4CAF50]">
            {currentResume ? '1/1' : '0/1'}
          </div>
          <div className="text-xs text-[#666666] mt-2">Free plan limit</div>
        </div>
      </div>

      {/* Main Content Card */}
      {currentResume ? (
        <div className="card">
          <div className="flex items-start gap-6">
            {/* Resume Thumbnail */}
            <div className="w-32 h-40 bg-[#F9FAFB] border-2 border-[#E0E0E0] rounded-md flex items-center justify-center">
              <FileText size={48} className="text-[#666666]" />
            </div>

            {/* Resume Info */}
            <div className="flex-1">
              <h3 className="text-xl font-bold text-[#333333] mb-2">
                Your Resume
              </h3>
              <div className="space-y-1 text-sm text-[#666666] mb-4">
                <div>‚Ä¢ Uploaded: {new Date(currentResume.created_at).toLocaleDateString()}</div>
                <div>‚Ä¢ Pages: {currentResume.page_count || 2}</div>
                <div>‚Ä¢ Size: 2.4 MB</div>
              </div>

              <button
                onClick={() => navigate(`/app/tailor/${currentResume.id}`)}
                className="btn-primary w-full sm:w-auto"
              >
                Tailor this Resume ‚Üí
              </button>
            </div>
          </div>
        </div>
      ) : (
        <div className="card text-center py-12">
          <Upload size={48} className="mx-auto text-[#666666] mb-4" />
          <h3 className="text-xl font-bold text-[#333333] mb-2">
            Upload Your Resume
          </h3>
          <p className="text-[#666666] mb-6">
            Get started by uploading your resume in PDF or DOCX format.
          </p>
          <button
            onClick={() => setShowUploadModal(true)}
            className="btn-primary"
          >
            Upload Resume
          </button>
        </div>
      )}

      {/* Upload Modal */}
      {showUploadModal && (
        <ResumeUpload
          isOpen={showUploadModal}
          onClose={() => setShowUploadModal(false)}
        />
      )}
    </div>
  )
}
```

**‚úÖ Test Checkpoint:**
- Dashboard renders with new design
- Stats cards display correctly
- Resume preview card shows if resume exists
- Upload CTA shows if no resume
- "Tailor this Resume" button navigates correctly
- Green (#4CAF50) color applied to CTAs

import Landing from './pages/Landing'

export default function App() {

    <Routes>

      {/* Redirect old dashboard route */}
      <Route path="/dashboard" element={<Navigate to="/app/dashboard" replace />} />
    </Routes>
  )
}

serve(async (req) => {

  })

    success_url: `${Deno.env.get('FRONTEND_URL')}/app/billing?success=true`,
    cancel_url: `${Deno.env.get('FRONTEND_URL')}/app/billing?canceled=true`,
    metadata: { supabase_user_id: userId }
  })

  return new Response(JSON.stringify({ url: session.url }), {
    headers: { 'Content-Type': 'application/json' }

### Step 7.3: Handle Stripe Webhooks (90 min)

**Files Created:**
- `backend/supabase/functions/stripe-webhook/index.ts`

**Events to Handle:**
1. `checkout.session.completed` - Activate subscription
2. `invoice.payment_succeeded` - Renew subscription
3. `invoice.payment_failed` - Suspend subscription
4. `customer.subscription.deleted` - Cancel subscription

**Implementation:**
```ts
import { serve } from 'https://deno.land/std@0.168.0/http/server.ts'
import Stripe from 'https://esm.sh/stripe@13.7.0'
import { createClient } from 'https://esm.sh/@supabase/supabase-js@2'

const stripe = new Stripe(Deno.env.get('STRIPE_SECRET_KEY') || '', {
  apiVersion: '2023-10-16',
})

serve(async (req) => {
  const signature = req.headers.get('stripe-signature')
  const body = await req.text()

  let event
  try {
    event = stripe.webhooks.constructEvent(
      body,
      signature!,
      Deno.env.get('STRIPE_WEBHOOK_SECRET')!
    )
  } catch (err) {
    return new Response(JSON.stringify({ error: 'Invalid signature' }), {
      status: 400
    })
  }

  const supabase = createClient(
    Deno.env.get('SUPABASE_URL')!,
    Deno.env.get('SUPABASE_SERVICE_ROLE_KEY')!
  )

  switch (event.type) {
    case 'checkout.session.completed': {
      const session = event.data.object as Stripe.Checkout.Session
      const userId = session.metadata?.supabase_user_id

      await supabase
        .from('user_profiles')
        .update({
          subscription_tier: 'pro',
          subscription_status: 'active',
          stripe_customer_id: session.customer as string,
          stripe_subscription_id: session.subscription as string,
          tailorings_limit: 999999, // Unlimited
          subscription_expires_at: null
        })
        .eq('user_id', userId)
      break
    }

    case 'customer.subscription.deleted': {
      const subscription = event.data.object as Stripe.Subscription
      await supabase
        .from('user_profiles')
        .update({
          subscription_tier: 'free',
          subscription_status: 'canceled',
          tailorings_limit: 1,
          tailorings_used: 0
        })
        .eq('stripe_subscription_id', subscription.id)
      break
    }
  }

  return new Response(JSON.stringify({ received: true }))
})
```

## ğŸ“ Implementation Notes

### Paywall Implementation Strategy
**Decision:** Focus on core functionality first, add paywall/monetization later

**Approach:**
1. Build all pages and features WITHOUT paywall enforcement initially
2. Mark paywall trigger points with `// TODO: PAYWALL` comments
3. Implement full feature set as if all users are "Pro"
4. Add usage tracking infrastructure last
5. Activate hard paywalls in Phase 6-7 (Day 6-7)

**Paywall Trigger Points to Mark:**
- [ ] Download 2nd+ tailored resume
- [ ] Upload 2nd resume (free plan = 1 resume limit)
- [ ] Add 6th+ job description (free plan = 5 jobs max)
- [ ] Access premium templates (if we decide to gate them)

**Benefits:**
- Faster development velocity
- Test full user flow without friction
- Iterate on features before constraining them
- Easier to adjust limits based on user feedback
